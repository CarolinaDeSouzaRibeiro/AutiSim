using System;
using UnityEditor;
using UnityEngine;

public class QuestSystem : MonoBehaviour
{
    public Transform cashierPoint; // The position of the cashier
    public Transform chairPoint; // The position of the chair
    public GameObject food; // The food object to interact with
    public GameObject player; // The player object

    private enum QuestState
    {
        WalkToCashier,
        MakeOrder,
        WalkToChair,
        EatFood,
        Completed
    }

    private QuestState currentQuestState;

    void Start()
    {
        currentQuestState = QuestState.WalkToCashier;
        Debug.Log("Quest started: Walk to the cashier.");
    }

    void Update()
    {
        switch (currentQuestState)
        {
            case QuestState.WalkToCashier:
                CheckPlayerPosition(cashierPoint.position, "Make your order.");
                break;

            case QuestState.MakeOrder:
                if (Input.GetKeyDown(KeyCode.E)) // Replace with VR interaction logic
                {
                    Debug.Log("Order made. Walk to the chair.");
                    currentQuestState = QuestState.WalkToChair;
                }
                break;

            case QuestState.WalkToChair:
                CheckPlayerPosition(chairPoint.position, "Eat your food.");
                break;

            case QuestState.EatFood:
                if (Input.GetKeyDown(KeyCode.E)) // Replace with VR interaction logic
                {
                    Debug.Log("Food eaten. Quest completed!");
                    currentQuestState = QuestState.Completed;
                }
                break;

            case QuestState.Completed:
                Debug.Log("You have completed the quest!");
                break;
        }
    }

    private void CheckPlayerPosition(Vector3 targetPosition, string nextTaskMessage)
    {
        float distance = Vector3.Distance(player.transform.position, targetPosition);
        if (distance < 1.5f) // Adjust the distance threshold as needed
        {
            Debug.Log(nextTaskMessage);
            currentQuestState++;
        }
    }
}
